function makeMoveUnitCommand(unit,x,y){
	// This function here is the command object:
	return function(){
		unit.moveTo(x,y);
	}
}

function makeMoveUnitCommand(unit,x,y){
	var xBefore,yBefore;
	return {		
		execute:function(){
			xBefore = unit.x();
			yBefore = unit.y();
			unit.moveTo(x,y);
		},
		undo:function(){
			unit.moveTo(xBefore,yBefore);
		}
	};
}
